# 3. Nomadic Intelligence: HybridDeltaTracker 구현
class NomadicTracker:
    def __init__(self, alpha=0.8, beta=0.85):
        self.prev_x = None
        self.ema_err = 0.0
        self.baseline_err = 0.0
        self.alpha = alpha  # 빠른 변화 감지
        self.beta = beta    # 느린 변화 감지 (베이스라인)

    def update(self, current_x, current_err):
        # 환경 변화 신호 (Delta x_env): 코사인 거리 활용
        if self.prev_x is None:
            delta_env = 0.0
        else:
            # 고차원 벡터의 방향 변화 측정
            cos_sim = F.cosine_similarity(current_x.flatten(), self.prev_x.flatten(), dim=0)
            delta_env = 1.0 - cos_sim.item()
        
        self.prev_x = current_x.detach().clone()

        # 예측 오차 신호 (Delta x_err): 상대적 오차 측정
        self.ema_err = self.alpha * self.ema_err + (1 - self.alpha) * current_err
        self.baseline_err = self.beta * self.baseline_err + (1 - self.beta) * current_err
        delta_err = max(0, self.ema_err - self.baseline_err)

        # 하이브리드 에너지원 (Delta x_hybrid)
        delta_hybrid = torch.tanh(torch.tensor(delta_env + delta_err)).item()
        
        return delta_env, delta_err, delta_hybrid

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
from collections import deque

# ==========================================
# 1. 확장된 Nomadic Tracker
# ==========================================
class AdvancedNomadicTracker:
    def __init__(self, alpha=0.8, beta=0.85, tau_min=2.0, tau_max=8.0, tau_window=8):
        self.alpha = alpha
        self.beta = beta
        self.tau_min = tau_min
        self.tau_max = tau_max
        self.tau_var_scale = 6.0
        self.tau_window = tau_window
        self.reset()

    def reset(self):
        self.prev_x = None
        self.ema_err = 0.0
        self.baseline_err = 0.0
        self.recent_delta_env = deque(maxlen=self.tau_window)
        print("✅ Advanced Tracker 초기화 완료 (σ² 및 τ_dynamic 추적 활성화)")

    def update(self, current_x, current_err):
        if self.prev_x is None:
            delta_env = 0.0
        else:
            cos_sim = F.cosine_similarity(current_x.flatten(), self.prev_x.flatten(), dim=0)
            delta_env = 1.0 - cos_sim.item()
        
        self.prev_x = current_x.detach().clone()
        self.recent_delta_env.append(delta_env)

        self.ema_err = self.alpha * self.ema_err + (1 - self.alpha) * current_err
        self.baseline_err = self.beta * self.baseline_err + (1 - self.beta) * current_err
        delta_err = max(0, self.ema_err - self.baseline_err)

        delta_hybrid = torch.tanh(torch.tensor(delta_env + delta_err)).item()

        sigma2_delta = float(np.var(self.recent_delta_env)) if len(self.recent_delta_env) >= 2 else 0.0
        dynamic_tau = self.tau_min + (self.tau_max - self.tau_min) / (1.0 + self.tau_var_scale * sigma2_delta)
        
        return delta_env, delta_err, delta_hybrid, sigma2_delta, dynamic_tau

# ==========================================
# 2. 경량화된 Policy Net (차원 오류 수정)
# ==========================================
class LightweightPolicyNet(nn.Module):
    def __init__(self, input_dim=1536, hidden_dim=64):
        super().__init__()
        self.proj = nn.Linear(input_dim, hidden_dim)
        self.shared = nn.Sequential(
            nn.Linear(hidden_dim + 5, hidden_dim), 
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
        )
        self.stay_switch_head = nn.Linear(hidden_dim, 2)
        self.mode_head = nn.Linear(hidden_dim, 2)

    def forward(self, x_summary, meta_signals):
        # x_summary가 (1, 1, dim)인 경우 (1, dim)으로 조정
        if x_summary.dim() == 3:
            x_summary = x_summary.squeeze(1)
            
        h_proj = F.relu(self.proj(x_summary))
        # h_proj와 meta_signals 모두 (Batch, Dim) 형태여야 함
        policy_input = torch.cat([h_proj, meta_signals], dim=-1)
        h = self.shared(policy_input)
        
        stay_switch_probs = F.softmax(self.stay_switch_head(h), dim=-1)
        mode_probs = F.softmax(self.mode_head(h), dim=-1)
        return stay_switch_probs, mode_probs
